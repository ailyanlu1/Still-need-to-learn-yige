# CLRS中的红黑树关键算法伪代码实现

# 左旋转代码

LF-RT(T, x)
  y = x.right
  x.right = y.left
  if y.left != T.nil
    y.left.parent = x
  y.parent = x.parent
  if x.parent == T.nil
    T.root = y
  elif x.parent.left == x
    x.parent.left = y
  else x.parent.right == x
    x.parent.right = y
  y.left = x
  x.parent = y

# 移植代码

RB-TRANSPLANT(T, u, v)
  if u.parent == T.nil
    T.root = v
  elif u.parent.left == u
    u.parent.left = v
  else 
    u.parent.right = v
  v.parent = u.parent
  
# 插入代码

RB-INSERT(T, z)
  x, y = T.root, T.nil
  while x != T.nil
    y = x
    if x.val < z.val
      x = x.right
    else
      x = x.left
  
  z.parent = y
  
  if y == T.nil
    T.root = z
  elif y.val < z.val
    y.right = z
  else
    y.left = z
  z.left = T.nil
  z.right = T.nil
  z.color = RED
  RB-INSERT-FIX-UP(T, z)
  
# 插入修正函数

RB-INSERT-FIX-UP(T, z)
  while z.parent.color == RED
    if z.parent == z.parent.parent.left
      y = z.parent.parent.right
      if y.color == RED
        y.color = BLACK
        z.parent = BLACK
        z.parent.parent = RED
        z = z.parent.parent
      elif y.color == BLACK
        if z == z.parent.right
          z = z.parent
          LF-RT(T, z)
        z.parent.color = BLACK
        z.parent.parent.color = RED
        RT-RT(T, z.parent.parent)
    else (The same as if statement, just change left and right for first two lines)
  T.root.color = BLACK
  
# 删除函数

RB-DELETE(T, z)
  y = z
  y-ori-color = z.color
  if z.left == T.nil
    x = z.right
    RB-TRANSPLANT(T, z, z.right)
  elif z.right == T.nil
    x = z.left
    RB-TRANSPLANT(T, z, z.left)
  else
    y = Tree-min(z.right)
    y-ori-color = y.color
    x = y.right
    if z.right == y
      x.parent = y
    else
      RB-TRANSPLANT(T, y, y.right)
      y.right = z.right
      y.right.parent = y
    RB-TRANSPLANT(T, z, y)
    y.left = z.left
    y.left.parent = y
    y.color = z.color
  if y-ori-color == BLACK
    RB-DELETE-FIXUP(T, x)
