# CLRS中的红黑树关键算法伪代码实现

# 左旋转代码

LF-RT(T, x)
  y = x.right
  x.right = y.left
  if y.left != T.nil
    y.left.parent = x
  y.parent = x.parent
  if x.parent == T.nil
    T.root = y
  elif x.parent.left == x
    x.parent.left = y
  else x.parent.right == x
    x.parent.right = y
  y.left = x
  x.parent = y
